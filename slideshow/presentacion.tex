\subsection{Objetivos}
\begin{frame}
\frametitle{Objetivos}
{\large
Integrar un sistema de visualización de datos en 3 dimensiones en un sistema complejo de visualización e interacción de datos\begin{block}{Sub-objetivos}
\begin{enumerate}
\item Construir algunas visualizaciones complejas en 3 dimensiones
\item Integrar escenas tridimensionales en Kibana
\item Integrar distintas escenas en un dashboard
\item Permitir filtrar datos mediante clicks en los objetos 3D
\item Ayudar a la mejora y extensión de la biblioteca gráfica 3D usada

\end{enumerate}
\end{block}
}
\end{frame}
